import React, { useEffect, useState } from "react";
import axios from "axios";

function StudentList() {
  const [students, setStudents] = useState([]);

  useEffect(() => {
    axios.get("http://127.0.0.1:8000/api/students/")
      .then(response => {
        setStudents(response.data);
      })
      .catch(error => {
        console.error("There was an error fetching the data!", error);
      });
  }, []);

  return (
    <div style={{ textAlign: 'center', marginTop: '30px' }}>
      <h2>ğŸ“‹ Student Records</h2>
      <table style={{ margin: 'auto', borderCollapse: 'collapse' }}>
        <thead>
          <tr style={{ backgroundColor: '#e3f2fd' }}>
            <th style={{ padding: '10px', border: '1px solid #90caf9' }}>Name</th>
            <th style={{ padding: '10px', border: '1px solid #90caf9' }}>Age</th>
            <th style={{ padding: '10px', border: '1px solid #90caf9' }}>Course</th>
          </tr>
        </thead>
        <tbody>
          {students.map((student, index) => (
            <tr key={index}>
              <td style={{ border: '1px solid #90caf9', padding: '8px' }}>{student.name}</td>
              <td style={{ border: '1px solid #90caf9', padding: '8px' }}>{student.age}</td>
              <td style={{ border: '1px solid #90caf9', padding: '8px' }}>{student.course}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

export default StudentList;
